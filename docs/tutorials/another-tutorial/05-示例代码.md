# 示例代码

## 基础示例

### 示例 1：简单使用

```javascript
import { Example } from 'example-package';

const example = new Example({
    apiKey: 'your-api-key'
});

const result = await example.process('Hello');
console.log(result);
```

## 进阶示例

### 示例 2：错误处理

```javascript
import { Example, ValidationError } from 'example-package';

const example = new Example({
    apiKey: 'your-api-key'
});

try {
    const result = await example.process('input');
    console.log('成功:', result);
} catch (error) {
    if (error instanceof ValidationError) {
        console.error('验证失败:', error.message);
    } else {
        console.error('其他错误:', error);
    }
}
```

### 示例 3：批量处理

```javascript
import { Example } from 'example-package';

const example = new Example({
    apiKey: 'your-api-key'
});

const inputs = ['input1', 'input2', 'input3'];

const results = await Promise.all(
    inputs.map(input => example.process(input))
);

console.log('所有结果:', results);
```

## 实际应用示例

### 示例 4：Web 应用集成

```javascript
// Express.js 示例
const express = require('express');
const { Example } = require('example-package');

const app = express();
const example = new Example({
    apiKey: process.env.API_KEY
});

app.post('/api/process', async (req, res) => {
    try {
        const result = await example.process(req.body.input);
        res.json({ success: true, data: result });
    } catch (error) {
        res.status(400).json({ success: false, error: error.message });
    }
});

app.listen(3000);
```

### 示例 5：React 组件

```jsx
import React, { useState } from 'react';
import { Example } from 'example-package';

function ProcessComponent() {
    const [input, setInput] = useState('');
    const [result, setResult] = useState(null);
    const [loading, setLoading] = useState(false);

    const example = new Example({
        apiKey: process.env.REACT_APP_API_KEY
    });

    const handleProcess = async () => {
        setLoading(true);
        try {
            const data = await example.process(input);
            setResult(data);
        } catch (error) {
            console.error('处理失败:', error);
        } finally {
            setLoading(false);
        }
    };

    return (
        <div>
            <input
                value={input}
                onChange={(e) => setInput(e.target.value)}
            />
            <button onClick={handleProcess} disabled={loading}>
                {loading ? '处理中...' : '处理'}
            </button>
            {result && <div>{JSON.stringify(result)}</div>}
        </div>
    );
}
```

## 最佳实践

### 1. 使用配置对象

```javascript
// ✅ 好的做法
const config = {
    apiKey: process.env.API_KEY,
    timeout: 5000,
    retries: 3
};
const example = new Example(config);

// ❌ 不好的做法
const example = new Example(process.env.API_KEY);
```

### 2. 错误处理

```javascript
// ✅ 好的做法
try {
    const result = await example.process(input);
} catch (error) {
    // 记录错误
    logger.error('处理失败', error);
    // 返回默认值或重试
}
```

### 3. 资源清理

```javascript
// ✅ 好的做法
const example = new Example(config);
try {
    await example.process(input);
} finally {
    example.cleanup();
}
```
